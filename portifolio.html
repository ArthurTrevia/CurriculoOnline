Programa de sorveteira em phython

print('Bem-vindo a sorveteria do Arthur Evaldo Trevia Guimarães')
print('--------------------CARDÁPIO----------------------')
print('------| Tamanho | Cupuaçu (CP) | Açaí (AC) |------')
print('------| P | R$10,00 | R$12,00 |------')
print('------| M | R$15,00 | R$17,00 |------')
print('------| G | R$19,00 | R$21,00 |------')
acumulador = 0 #variável com valor inicial para somar o preço dos produtos
while True:
sabor = input('Entre com o sabor desejado (CP/AC):')
sabor = sabor.upper() #deixa a resposta padronizada como upper
if sabor != 'CP' and sabor != 'AC' and sabor != 'cp' and sabor != 'ac':
print('Sabor inválido. Digite o sabor corretamente!')
continue
tamanho = input('Entre com o tamanho desejado (P/M/G):')
tamanho = tamanho.upper() #deixa a resposta padronizada como upper
if tamanho != 'P' and tamanho != 'M' and tamanho != 'G' and tamanho != 'p' and tamanho != 'm' and tamanho != 'g':
print('Tamanho inválido. Digite o tamanho corretamente!')
continue
if sabor == 'CP' and tamanho == 'P':
print('Você escolheu o sabor Cupuaçu de tamanho Pequeno!')
acumulador = acumulador + 10
elif sabor == 'CP' and tamanho == 'M':
print('Você escolheu o sabor Cupuaçu de tamanho M!')
acumulador = acumulador + 15
elif sabor == 'CP' and tamanho == 'G':
print('Você escolheu o sabor Cupuaçu de tamanho G!')
acumulador = acumulador + 19
elif sabor == 'AC' and tamanho == 'P':
print('Você escolheu o sabor Açaí de tamanho P!')
acumulador = acumulador + 12
elif sabor == 'AC' and tamanho == 'M':
print('Você escolheu o sabor Açaí de tamanho M!')
acumulador = acumulador + 17
elif sabor == 'AC' and tamanho == 'G':
print('Você escolheu o sabor Açaí de tamanho G!')
acumulador = acumulador + 21
pedir_mais = input('Deseja pedir mais alguma coisa (S/outra tecla)?')
pedir_mais = pedir_mais.upper()
if pedir_mais == 'S':
continue #continua o programa caso digite S para pedir mais coisas
else: print('O valor a ser pago é de:R${:.2f}'.format(acumulador))
break #encerra o programa caso digite outra tecla diferente de S
